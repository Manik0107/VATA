class LinearRegressionExample(Scene):
    def construct(self):
        # Setup
        self.camera.background_color = "#1e1e1e"
        
        # Create axes
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 15, 2],
            x_length=8,
            y_length=6,
            axis_config={"color": BLUE}
        ).shift(DOWN*0.5)
        
        # Labels
        x_label = axes.get_x_axis_label("Size")
        y_label = axes.get_y_axis_label("Price")
        
        # Data points
        data = [(2, 4), (4, 7), (6, 9), (8, 12)]
        dots = VGroup(*[
            Dot(axes.coords_to_point(x, y), color=CYAN)
            for x, y in data
        ])
        
        # Regression line
        line = axes.plot(lambda x: 1.5*x + 1, x_range=[1, 9], color=RED)
        
        # Animate
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.wait(1)
        self.play(FadeIn(dots, lag_ratio=0.1))
        self.wait(1)
        self.play(Create(line))
        self.wait(2)
        
        # Cleanup
        self.play(
            FadeOut(axes),
            FadeOut(x_label),
            FadeOut(y_label),
            FadeOut(dots),
            FadeOut(line)
        )
